\begin{frame}[fragile]{The AMPL Model - Parameters and Variables}
\begin{minted}[fontsize=\small, breaklines, frame=single, autogobble]{text}
set Car;
set Variables;

param y {i in Car};
param x {i in Car, j in Variables};
param t;

var bplus{j in Variables}  >= 0;
var bminus{j in Variables} >= 0;
\end{minted}
\end{frame}

\begin{frame}[fragile]{The AMPL Model - Obective Function and Constraints}
\begin{minted}[fontsize=\small, breaklines, frame=single, autogobble]{text}
minimize SSE:
    sum {i in Car}
        ( y[i] - sum {j in Variables} (bplus[j] - bminus[j]) * x[i,j] )^2;

subject to L1_budget:
    sum {j in Variables: j != 'intercept'} (bplus[j] + bminus[j]) <= t;
\end{minted}
\end{frame}
